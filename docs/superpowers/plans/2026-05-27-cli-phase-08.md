Create `cli/client.py`:

```python
from __future__ import annotations

from typing import Any

import httpx

DEFAULT_API_BASE = "http://127.0.0.1:8000"


class ApiRequestError(RuntimeError):
    def __init__(self, status_code: int, message: str, body: object) -> None:
        super().__init__(message)
        self.status_code = status_code
        self.body = body


class HttpClient:
    def __init__(
        self,
        api_base: str = DEFAULT_API_BASE,
        *,
        transport: httpx.BaseTransport | None = None,
    ) -> None:
        self._client = httpx.Client(
            base_url=api_base.rstrip("/"),
            transport=transport,
            timeout=10.0,
        )

    def __enter__(self) -> "HttpClient":
        return self

    def __exit__(self, *_: object) -> None:
        self._client.close()

    def request(self, method: str, path: str, *, json: object | None = None) -> Any:
        try:
            response = self._client.request(method, path, json=json)
        except httpx.HTTPError as exc:
            raise RuntimeError(f"Request failed: {exc}") from exc
        body = response.json() if response.content else None
        if response.is_error:
            message = body.get("detail") if isinstance(body, dict) else None
            raise ApiRequestError(response.status_code, message or f"Request failed: {response.status_code}", body)
        return body

    def get(self, path: str) -> Any:
        return self.request("GET", path)

    def post(self, path: str, payload: object) -> Any:
        return self.request("POST", path, json=payload)

    def patch(self, path: str, payload: object) -> Any:
        return self.request("PATCH", path, json=payload)
```

Create `cli/output.py`:

```python
from __future__ import annotations

import json
from collections.abc import Iterable, Mapping
from typing import Any


def print_result(value: Any, *, as_json: bool) -> None:
    if as_json:
        print(json.dumps(value, ensure_ascii=False, indent=2))
        return
    if isinstance(value, list):
        for row in value:
            print(_format_row(row))
        if not value:
            print("No results.")
        return
    if isinstance(value, Mapping):
        print(_format_row(value))
        return
    print(value)


def _format_row(row: Mapping[str, Any]) -> str:
    keys: Iterable[str] = ("identifier", "id", "name", "title", "type", "status", "priority", "updatedAt")
    return "  ".join(f"{key}={row[key]}" for key in keys if row.get(key) is not None)
```

from __future__ import annotations

import sys

from .client import ApiRequestError, HttpClient
from .output import print_result
from .parser import build_parser


def main(argv: list[str] | None = None) -> int:
    args = build_parser().parse_args(argv)
    try:
        with HttpClient(args.api_base) as client:
            result = args.handler(client, args)
        print_result(result, as_json=args.as_json)
        return 0
    except (ApiRequestError, RuntimeError, ValueError) as exc:
        print(str(exc), file=sys.stderr)
        return 1


if __name__ == "__main__":
    raise SystemExit(main())

Create `cli/commands/organization.py`:

```python
from __future__ import annotations

import argparse
from typing import Any

from cli.client import HttpClient


def _payload(**values: object) -> dict[str, object]:
    return {key: value for key, value in values.items() if value is not None}


def list_organizations(client: HttpClient, _: argparse.Namespace) -> Any:
    return client.get("/api/orgs")


def get_organization(client: HttpClient, args: argparse.Namespace) -> Any:
    return client.get(f"/api/orgs/{args.org_id}")


def create_organization(client: HttpClient, args: argparse.Namespace) -> Any:
    return client.post("/api/orgs", _payload(name=args.name, description=args.description))


def update_organization(client: HttpClient, args: argparse.Namespace) -> Any:
    return client.patch(
        f"/api/orgs/{args.org_id}",
        _payload(name=args.name, description=args.description, brandColor=args.brand_color),
    )


def register_organization_parser(subcommands: argparse._SubParsersAction[argparse.ArgumentParser]) -> None:
    root = subcommands.add_parser("organization")
    commands = root.add_subparsers(dest="operation", required=True)
    list_parser = commands.add_parser("list")
    list_parser.set_defaults(handler=list_organizations)
    get_parser = commands.add_parser("get")
    get_parser.add_argument("org_id")
    get_parser.set_defaults(handler=get_organization)
    create_parser = commands.add_parser("create")
    create_parser.add_argument("--name", required=True)
    create_parser.add_argument("--description")
    create_parser.set_defaults(handler=create_organization)
    update_parser = commands.add_parser("update")
    update_parser.add_argument("org_id")
    update_parser.add_argument("--name")
    update_parser.add_argument("--description")
    update_parser.add_argument("--brand-color")
    update_parser.set_defaults(handler=update_organization)
```

Modify `cli/parser.py` to import `register_organization_parser`, call it in place of `subcommands.add_parser("organization")`, and leave the bare issue and approval parsers until their tasks are implemented.
