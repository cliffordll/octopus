```ts
export class ApiError extends Error {
  constructor(
    public readonly status: number,
    message: string,
    public readonly body: unknown,
  ) {
    super(message);
    this.name = "ApiError";
  }
}

async function request<T>(path: string, init: RequestInit = {}): Promise<T> {
  const headers = new Headers(init.headers);
  if (init.body && !headers.has("Content-Type")) headers.set("Content-Type", "application/json");
  const response = await fetch(path, { ...init, headers, method: init.method ?? "GET" });
  const body = await response.json().catch(() => null) as unknown;
  if (!response.ok) {
    const message =
      typeof body === "object" && body !== null && "detail" in body && typeof body.detail === "string"
        ? body.detail
        : `Request failed: ${response.status}`;
    throw new ApiError(response.status, message, body);
  }
  return body as T;
}

export const api = {
  get: <T>(path: string) => request<T>(path, { method: "GET" }),
  post: <T>(path: string, body: unknown) => request<T>(path, { method: "POST", body: JSON.stringify(body) }),
  patch: <T>(path: string, body: unknown) => request<T>(path, { method: "PATCH", body: JSON.stringify(body) }),
};
```

```ts
import { api } from "./client";
import type { IssueComment, IssueDetail, IssueListItem, IssuePriority, IssueReviewDecision, IssueStatus } from "./types";

export interface IssueFilters {
  status?: IssueStatus;
  assigneeAgentId?: string;
  projectId?: string;
  originKind?: "manual" | "automation_execution";
  originId?: string;
}

export const issuesApi = {
  list(orgId: string, filters: IssueFilters = {}) {
    const params = new URLSearchParams();
    Object.entries(filters).forEach(([key, value]) => value && params.set(key, value));
    const suffix = params.size ? `?${params.toString()}` : "";
    return api.get<IssueListItem[]>(`/api/orgs/${orgId}/issues${suffix}`);
  },
  get: (id: string) => api.get<IssueDetail>(`/api/issues/${id}`),
  create: (orgId: string, payload: { title: string; description?: string | null; status?: IssueStatus; priority?: IssuePriority }) =>
    api.post<IssueDetail>(`/api/orgs/${orgId}/issues`, payload),
  update: (id: string, payload: Partial<Pick<IssueDetail, "title" | "description" | "status" | "priority">>) =>
    api.patch<IssueDetail>(`/api/issues/${id}`, payload),
  comments: (id: string) => api.get<IssueComment[]>(`/api/issues/${id}/comments`),
  comment: (id: string, body: string) => api.post<IssueComment>(`/api/issues/${id}/comments`, { body }),
  review: (id: string, payload: { decision: IssueReviewDecision; note?: string | null }) =>
    api.post<IssueDetail>(`/api/issues/${id}/review-decision`, payload),
};
```
